```python
from dataclasses import dataclass, field
from datetime import datetime
from typing import List, Optional

@dataclass
class Task:
    """タスククラス"""
    id: str
    name: str
    deliverable_id: str
    start_date: datetime
    end_date: datetime
    status: str  # '完了', '着手中', '未着手'
    
    @property
    def progress(self) -> float:
        """ステータスに基づいた進捗率を返す (0-100)"""
        if self.status == '完了':
            return 100.0
        elif self.status == '着手中':
            return 50.0
        else:  # '未着手'
            return 0.0
    
    @property
    def duration_days(self) -> int:
        """タスク期間（日数）"""
        return (self.end_date - self.start_date).days + 1
    
    @property
    def days_delayed(self) -> int:
        """遅延日数を計算"""
        if self.status != '完了' and datetime.now() > self.end_date:
            return (datetime.now() - self.end_date).days
        return 0
    
    @property
    def is_delayed(self) -> bool:
        """遅延判定"""
        return self.days_delayed > 0

@dataclass
class Deliverable:
    """成果物クラス"""
    id: str
    name: str
    theme_id: str
    start_date: datetime
    end_date: datetime
    tasks: List[Task] = field(default_factory=list)
    
    @property
    def status(self) -> str:
        """含まれるタスクのステータスから判定"""
        if not self.tasks:
            return '未着手'
        completed = sum(1 for t in self.tasks if t.status == '完了')
        in_progress = sum(1 for t in self.tasks if t.status == '着手中')
        if completed == len(self.tasks):
            return '完了'
        elif in_progress > 0 or completed > 0:
            return '着手中'
        else:
            return '未着手'
    
    @property
    def overall_progress(self) -> float:
        """期間加重平均で進捗率を計算"""
        if not self.tasks:
            return 0.0
        total_duration = sum(t.duration_days for t in self.tasks)
        if total_duration == 0:
            return 0.0
        weighted_progress = sum(t.progress * t.duration_days for t in self.tasks)
        return (weighted_progress / total_duration)
    
    @property
    def planned_progress(self) -> float:
        """計画進捗（スケジュール進捗）"""
        total_days = (self.end_date - self.start_date).days
        elapsed_days = (datetime.now() - self.start_date).days
        if total_days <= 0:
            return 0.0
        return min(100.0, max(0.0, (elapsed_days / total_days) * 100))

@dataclass
class Theme:
    """業務テーマクラス"""
    id: str
    name: str
    start_date: datetime
    end_date: datetime
    deliverables: List[Deliverable] = field(default_factory=list)
    
    @property
    def status(self) -> str:
        """ステータス判定"""
        if self.overall_progress >= self.planned_progress:
            return 'On Track'
        elif self.overall_progress >= self.planned_progress - 10:
            return 'At Risk'
        else:
            return 'Delayed'
    
    @property
    def overall_progress(self) -> float:
        """期間加重平均で進捗率を計算"""
        all_tasks = [task for d in self.deliverables for task in d.tasks]
        if not all_tasks:
            return 0.0
        total_duration = sum(t.duration_days for t in all_tasks)
        if total_duration == 0:
            return 0.0
        weighted_progress = sum(t.progress * t.duration_days for t in all_tasks)
        return (weighted_progress / total_duration)
    
    @property
    def planned_progress(self) -> float:
        """計画進捗（スケジュール進捗）"""
        total_days = (self.end_date - self.start_date).days
        elapsed_days = (datetime.now() - self.start_date).days
        if total_days <= 0:
            return 0.0
        return min(100.0, max(0.0, (elapsed_days / total_days) * 100))

@dataclass
class DashboardData:
    """ダッシュボード用データクラス"""
    themes: List[Theme]
    generated_at: datetime
    
    @property
    def overall_progress(self) -> float:
        """全体進捗"""
        if not self.themes:
            return 0.0
        return sum(t.overall_progress for t in self.themes) / len(self.themes)
    
    @property
    def overall_planned_progress(self) -> float:
        """全体計画進捗"""
        if not self.themes:
            return 0.0
        return sum(t.planned_progress for t in self.themes) / len(self.themes)
    
    @property
    def delayed_items(self) -> List:
        """遅延アイテム一覧"""
        delayed = []
        for theme in self.themes:
            if theme.overall_progress < theme.planned_progress:
                delayed.append(('theme', theme))
            for deliv in theme.deliverables:
                if deliv.overall_progress < deliv.planned_progress:
                    delayed.append(('deliverable', deliv))
                for task in deliv.tasks:
                    if task.is_delayed:
                        delayed.append(('task', task))
        return delayed
    
    @property
    def in_progress_items(self) -> List:
        """進行中アイテム一覧"""
        in_progress = []
        for theme in self.themes:
            if theme.status == 'In Progress':
                in_progress.append(('theme', theme))
            for deliv in theme.deliverables:
                if deliv.status == 'In Progress':
                    in_progress.append(('deliverable', deliv))
                for task in deliv.tasks:
                    if task.status == 'In Progress':
                        in_progress.append(('task', task))
        return in_progress
```

```python
import openpyxl
from typing import List, Dict
from datetime import datetime

class ExcelReader:
    """Excelファイル読み込みクラス"""
    
    def __init__(self, filepath: str):
        self.filepath = filepath
        self.workbook = None
        self.worksheet = None
    
    def read(self) -> List[Dict]:
        """Excelファイルを読み込み、行データのリストを返す"""
        self.workbook = openpyxl.load_workbook(self.filepath)
        self.worksheet = self.workbook.active
        
        rows = []
        headers = None
        
        for idx, row in enumerate(self.worksheet.iter_rows(values_only=True)):
            if idx == 0:
                headers = row
                continue
            
            row_dict = {}
            for i, header in enumerate(headers):
                if i < len(row):
                    row_dict[header] = row[i]
            rows.append(row_dict)
        
        self.workbook.close()
        return rows
    
    def validate(self, rows: List[Dict]) -> bool:
        """バリデーション"""
        required_fields = ['業務テーマ', '成果物', 'タスク', '開始時期', '終了時期']
        for field in required_fields:
            if not any(field in row for row in rows):
                raise ValueError(f"Required field '{field}' not found")
        return True
```

```python
from typing import List, Dict
from datetime import datetime
from .models import Theme, Deliverable, Task
import uuid

class WBSParser:
    """WBSデータパーサー"""
    
    def __init__(self):
        self.themes = []
        self.current_theme = None
        self.current_deliverable = None
    
    def parse(self, rows: List[Dict]) -> List[Theme]:
        """WBS形式のデータをパース"""
        self.themes = []
        
        for row in rows:
            theme_name = row.get('業務テーマ', '').strip()
            deliv_name = row.get('成果物', '').strip()
            task_name = row.get('タスク', '').strip()
            
            start_date = self._parse_date(row.get('開始時期'))
            end_date = self._parse_date(row.get('終了時期'))
            status = row.get('ステータス', '未着手')  # デフォルトは未着手
            
            # Theme処理
            if theme_name:
                self.current_theme = Theme(
                    id=str(uuid.uuid4()),
                    name=theme_name,
                    start_date=start_date,
                    end_date=end_date,
                    deliverables=[]
                )
                self.themes.append(self.current_theme)
                self.current_deliverable = None
            
            # Deliverable処理
            if deliv_name and self.current_theme:
                self.current_deliverable = Deliverable(
                    id=str(uuid.uuid4()),
                    name=deliv_name,
                    theme_id=self.current_theme.id,
                    start_date=start_date,
                    end_date=end_date,
                    tasks=[]
                )
                self.current_theme.deliverables.append(self.current_deliverable)
            
            # Task処理
            if task_name and self.current_deliverable:
                task = Task(
                    id=str(uuid.uuid4()),
                    name=task_name,
                    deliverable_id=self.current_deliverable.id,
                    start_date=start_date,
                    end_date=end_date,
                    status=status
                )
                self.current_deliverable.tasks.append(task)
        
        return self.themes
    
    @staticmethod
    def _parse_date(date_str) -> datetime:
        """日付文字列をdatetimeに変換"""
        if isinstance(date_str, datetime):
            return date_str
        if date_str is None:
            return datetime.now()
        return datetime.strptime(str(date_str), '%Y-%m-%d')
```

```python
from datetime import datetime
from typing import List
from .models import Theme, Deliverable, Task

class ProgressCalculator:
    """進捗率計算クラス"""
    
    @staticmethod
    def calculate_planned_progress(start_date: datetime, end_date: datetime) -> float:
        """計画進捗を計算（スケジュール進捗）"""
        today = datetime.now()
        total_days = (end_date - start_date).days
        elapsed_days = (today - start_date).days
        
        if total_days <= 0:
            return 0.0
        
        progress = (elapsed_days / total_days) * 100
        return min(100.0, max(0.0, progress))
    
    @staticmethod
    def calculate_actual_progress(tasks: List[Task]) -> float:
        """実績進捗を計算（期間加重平均）
        
        タスク進捗率 × タスク期間 の合計 / 全タスク期間の合計
        
        タスク進捗率:
          - '完了': 100%
          - '着手中': 50%
          - '未着手': 0%
        """
        if not tasks:
            return 0.0
        
        total_duration = sum(t.duration_days for t in tasks)
        if total_duration == 0:
            return 0.0
        
        weighted_progress = sum(t.progress * t.duration_days for t in tasks)
        return (weighted_progress / total_duration)
```

```python
        """進行中アイテムリストHTML生成"""
        if not data.in_progress_items:
            return '<div class="list-item">No in-progress items</div>'
        
        html = ""
        for item_type, item in data.in_progress_items:
            if item_type == 'task':
                html += f'<div class="list-item">{item.name} | {item.progress:.0f}%</div>'
        return html
    
    @staticmethod
    def _generate_theme_rows(data: DashboardData) -> str:
        """テーマ行HTML生成"""
        html = ""
        for theme in data.themes:
            status_class = f"status-{theme.status.lower().replace(' ', '-')}"
            trend = "↑" if theme.overall_progress >= theme.planned_progress else "→" if theme.overall_progress >= theme.planned_progress - 10 else "↓"
            html += f"""
            <tr>
                <td>{theme.name}</td>
                <td><span class="status-badge {status_class}">{theme.status}</span></td>
                <td>{theme.overall_progress:.0f}%</td>
                <td>{trend}</td>
                <td><button>[詳細]</button></td>
            </tr>
            """
        return html
```

```python
from pathlib import Path
from datetime import datetime
from data.excel_reader import ExcelReader
from data.wbs_parser import WBSParser
from data.models import DashboardData
from presentation.html_generator import HTMLGenerator

def main(input_file: str, output_file: str = None):
    """メイン処理"""
    
    if output_file is None:
        output_file = Path(input_file).stem + "_dashboard.html"
    
    print(f"Reading Excel file: {input_file}")
    reader = ExcelReader(input_file)
    rows = reader.read()
    
    print(f"Parsing WBS structure...")
    parser = WBSParser()
    themes = parser.parse(rows)
    
    print(f"Generating dashboard data...")
    dashboard_data = DashboardData(
        themes=themes,
        generated_at=datetime.now()
    )
    
    print(f"Generating HTML...")
    html = HTMLGenerator.generate(dashboard_data)
    
    with open(output_file, 'w', encoding='utf-8') as f:
        f.write(html)
    
    print(f"Dashboard generated: {output_file}")

if __name__ == '__main__':
    import sys
    if len(sys.argv) < 2:
        print("Usage: python main.py <excel_file> [output_file]")
        sys.exit(1)
    
    input_file = sys.argv[1]
    output_file = sys.argv[2] if len(sys.argv) > 2 else None
    main(input_file, output_file)
```
